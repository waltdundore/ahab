test-unit:
	$(call print_header,Running Unit Tests)
	@if [ -d "tests/unit" ] && [ -n "$$(ls -A tests/unit/*.sh 2>/dev/null)" ]; then \
		for test in tests/unit/*.sh; do \
			$(call show_command,$$test,Unit test); \
			bash $$test || exit 1; \
		done; \
		echo ""; \
		$(call print_success,All unit tests passed); \
	else \
		echo "â„¹ No unit tests found (tests/unit/ is empty or missing)"; \
	fi
